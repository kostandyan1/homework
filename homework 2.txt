#include <stdio.h>

int main() {
    char number;
    int numBits = sizeof(char) * 8; // Total number of bits in char type
    int halfNumBits = numBits / 2;   // Half of the number of bits
    int simetrik = 1;
    
    printf("Enter a char type number: ");
    scanf("%s", &number);
    int i = 0;
    while(i < halfNumBits && ((number >> i) & 1) == ((number >> (numBits - 1 - i)) & 1))
            i++;
    if (i != halfNumBits)
        simetrik = 0;
    if (simetrik) {
        printf("True\n");
    } else {
        printf("False\n");
    }
    return 0;
}
